### 异步组件
```javascript
Vue.component('Foo', () => {
  component: delay(4000, () => ('Foo.js')),
  loading: LoadingComponent,
 	error: ErrorComponent,
  timeout: 5000
})

const delay = (time, fn) => {
  return new Promise((res, rej) => {
    setTimeout(() => {
      resolve(fn())
    }, time)
  })
}

const LoadingComponent = {
  template: `<div>Loading...</div>`
}

const ErrorComponent = {
  template: `<div>Error</div>`
}
```
